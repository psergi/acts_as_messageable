  def deliver(mailbox_type = :inbox, clean = true)
    self.clean() if clean
    self.recipients.each do |r|
      r.mailbox[mailbox_type] << self
    end
    self.on_deliver_callback.call(self, mailbox_type) unless self.on_deliver_callback.nil?
  end
